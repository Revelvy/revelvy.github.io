Java 反射

Java
Java
发布日期:   2020-02-02
文章字数:   1.9k
阅读时长:   9 分

引言

Reflection，代码在运行时可以借助反射 API 获取任何类的信息，并能直接操作任何对象的属性和方法（如private）。

正常：引入包 –> new实例化对象 –> 得到对象
反射：实例化对象 –> getClass() –> 得到完整包名 【通过镜子可以看到类的完整结构信息】

Class c = Class.forName(“java.lang.String”);
加载完类之后，在方法区中产生一个Class类对象（一个类只有一个Class类对象），包含类完整的结构信息。

当加载 User 字节码时，Class就会根据这个字节码的特征创建一个对象，对字节码进行描述

可以用 Class 类获取运行中的对象的类的信息，就能直接操作其属性和方法。

优点：动态灵活地创建对象和编译。
缺点：比 new 出来的对象性能降低。

编译 解释 动态 静态语言

动态：代码在运行时结构可变或变量类型可变（JS、PHP、Python）。
静态：代码在运行时结构不可变或变量类型不可变（C++）。
准动态：利用反射获得类似动态语言的特性 –> 灵活（Java）。

编译型：将源代码编译成字节码文件，代码安全度高，执行效率低。

解释型：解析器可以直接运行源代码，代码安全性低，执行效率高。

反射 API

  • java.lang.Class 类
  • java.lang.reflect.Method 类方法
  • java.lang.reflect.Field 类成员变量
  • java.lang.reflect.Constructor 类构造器

Class 类

Class对象只能由系统创建，Class类是反射根源，针对任何向动态加载、运行的类、必须先获得相应Class对象
一个加载的类在JVM中只有一个Class实例
一个Class对象对应一个加载到JVM中的一个.class文件
每个类的实例对象都会记得自己是由哪个Class实例所生成的（张三会记得Persion类）
通过Class可以完整地得到一个类中所有被加载的结构

Class 类方法

返回值 方法名
static 类<?> forName(String className) 返回指定类名的Class对象
T newInstance() 返回当前Class对象的一个实例
String getName() 返回当前Class对象表示的实体名
Class getSuperclass() 返回当前Class对象的父类Class对象
Class[] getInterfaces() 返回当前Class对象的接口
ClassLoader getClassLoader() 返回当前Class对象的类加载器
Constructor[] getConstructors() 返回包含构造器的数组
Method getMethod(String name,Clss ...T) 返回Method对象
getModifiers() 返回当前Class对象的类修饰符
getPackage() 返回当前Class对象的包名
Field[] getDeclaredFields() 返回包含Field对象的数组 
// 类修饰符
//   public,protected,private,final,static,abstract,interface
//   2的次幂数进行累加得到的结果是唯一的，通过一个值可以知道多个修饰符
System.out.println(clazz01.getModifiers());  //16
// 是否为原始数据类型（8种基本数据类型+void）
System.out.println(clazz01.isPrimitive());  //false
System.out.println(Integer.class.isPrimitive());  //false
System.out.println(int[].class.isPrimitive());  //false
System.out.println(int.class.isPrimitive());  //true
System.out.println(Integer.TYPE.isPrimitive());  //true

public class Hello03Method {

        Method[] dms = clazz.getDeclaredMethods();

     * 调用方法

通过反射调用注解

public class Hello04Reflection {
    public static void main(String[] args) throws ClassNotFoundException {
        //获取类对象
        Class clazz = Class.forName("com.xxxx.Dog");
        //获取类对象的所有注解
        Annotation[] annotations = clazz.getAnnotations();
        for (Annotation anno : annotations) {
            System.out.println(anno.annotationType());
            System.out.println(anno.value());
		   System.out.println(anno.type());
        }
        //获取类对象的指定注解
        Pet petanno = (Pet) clazz.getAnnotation(Pet.class);
        System.out.println(petanno.annotationType());
        System.out.println(petanno.value());
        System.out.println(petanno.type());
    }
}

@Target(ElementType.TYPE)
@Retention(RetentionPolicy.RUNTIME)
public @interface Pet {
    String value() default "家养宠物犬";
    String type();
}

@Target(ElementType.TYPE)
@Retention(RetentionPolicy.RUNTIME)
public @interface Shepherd {
    String value() default "";
    String type();
}

@Pet(type = "吉娃娃")

public class Hello05Test {

    public static void main(String[] args) throws ClassNotFoundException {
        //创建对象
        Teacher teacher = new Teacher(1, "张三丰", new Date(), "武当山", new Date());
        //声明一个可变字符串保存打印信息
        StringBuilder sb = new StringBuilder();
        //获取类对象
        Class clazz = Class.forName("com.xxxx.Teacher");
        //获取所有的属性
        Field[] dfs = clazz.getDeclaredFields();
        for (Field field : dfs) {
            //开始获取属性对应的注解
            FieldChineseName anno = field.getAnnotation(FieldChineseName.class);
            //判断是否为空
            if (anno != null) {
                try {
                    //获取取值方法  "getid" -> "getId"
                    String methodName = "get" + field.getName().toUpperCase().substring(0, 1) + field.getName().substring(1);
                    Method method = clazz.getMethod(methodName);  //public java.lang.Integer com.xxxx.Teacher.getId()
                    //获取中文名字  ,属性[教师编号]的值为[1]
                    sb.append(",属性[" + anno.chineseName() + "]的值为[" + method.invoke(teacher) + "]");
                } catch (NoSuchMethodException e) {
                    e.printStackTrace();
                } catch (IllegalAccessException e) {
                    e.printStackTrace();
                } catch (InvocationTargetException e) {
                    e.printStackTrace();
                }
            }
        }
//        System.out.println(sb);  //,属性[教师编号]的值为[1],属性[教师姓名]的值为[张三丰]
        System.out.println(sb.substring(1));  //属性[教师编号]的值为[1],属性[教师姓名]的值为[张三丰]
    }

}

@Target(ElementType.FIELD)
@Retention(RetentionPolicy.RUNTIME)
public @interface FieldChineseName {
    public String value() default "";

    public String chineseName();

}

class Teacher implements Serializable {
    @FieldChineseName(chineseName = "教师编号")
    private Integer id;
    @FieldChineseName(chineseName = "教师姓名")
    private String name;
    private Date birthday;
    private String address;
    private Date createTime;

    public Teacher() {
    }

    public Teacher(Integer id, String name, Date birthday, String address, Date createTime) {
        this.id = id;
        this.name = name;
        this.birthday = birthday;
        this.address = address;
        this.createTime = createTime;
    }

    public Integer getId() {
        return id;
    }

    public void setId(Integer id) {
        this.id = id;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public Date getBirthday() {
        return birthday;
    }

    public void setBirthday(Date birthday) {
        this.birthday = birthday;
    }

    public String getAddress() {
        return address;
    }

    public void setAddress(String address) {
        this.address = address;
    }

    public Date getCreateTime() {
        return createTime;
    }

    public void setCreateTime(Date createTime) {
        this.createTime = createTime;
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (o == null || getClass() != o.getClass()) return false;
        Teacher teacher = (Teacher) o;
        return Objects.equals(id, teacher.id) &&
                Objects.equals(name, teacher.name) &&
                Objects.equals(birthday, teacher.birthday) &&
                Objects.equals(address, teacher.address) &&
                Objects.equals(createTime, teacher.createTime);
    }

    @Override
    public int hashCode() {
        return Objects.hash(id, name, birthday, address, createTime);
    }

    @Override
    public String toString() {
        return "Teacher{" +
                "id=" + id +
                ", name='" + name + '\'' +
                ", birthday=" + birthday +
                ", address='" + address + '\'' +
                ", createTime=" + createTime +
                '}';
    }
}
