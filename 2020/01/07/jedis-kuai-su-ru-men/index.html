        multi.set("k2", result);
        multi.exec();

        multi.discard();
        e.printStackTrace();

        System.out.println(jedis.get("k1"));  // {"name":"kuangshen","hello":"world"}
        System.out.println(jedis.get("k2"));  // {"name":"kuangshen","hello":"world"}
        jedis.close();
    }
}
