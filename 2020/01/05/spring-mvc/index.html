引言

MVC

🌠 概念：模型 - 视图 - 控制器，是以设计界面应用程序为基础的设计思想。

  • M：封装应用程序数据。
  • V：展示数据。
  • C：接收请求，并调用后台业务层来处理业务逻辑，业务层返回数据，控制器接收数据和准备模型在 V 层展示。

Spring MVC

🌠 概念：基于Java的实现了Web MVC设计思想的请求驱动类型的轻量级Web框架
🎉 思想：通过分离 MVC 在应用程序中的角色，对 Web 层进行解耦，将业务逻辑从界面中分离出来。

流程

  1. 用户发送请求,请求被调度器（DispatherServlet）捕获；
  2. 调度器对请求URL解析获取URI,根据URI,调用处理器映射器HandlerMapping；
  3. 调度器获得返回的HandlerExecutionChain（包括处理器对象Handler以及对应的拦截器）；
  4. 调度器根据获得的HandlerExecutionChain，选择一个合适的处理器适配器HandlerAdapter。
    （附注：如果成功获得HandlerAdapter后，此时将开始执行拦截器的preHandler(…)方法）
  5. 处理器适配器根据请求的处理器适配并执行对应的处理器；
    HandlerAdapter(提取Request中的模型数据，填充Handler入参，开始执行Handler（Controller)。
    在填充Handler的入参过程中，根据配置，Spring将做一些额外的工作：HttpMessageConveter：
    将请求消息（如Json、xml等数据）转换成一个对象，将对象转换为指定的响应信息。
    数据转换：对请求消息进行数据转换。如String转换成Integer、Double等数据格式化：
    数据格式化。 如将字符串转换成格式化数字或格式化日期等
    数据验证： 验证数据的有效性（长度、格式等），验证结果存储到BindingResult或Error中）
  6. 处理器执行完毕，返回一个ModelAndView给处理器适配器
  7. 处理器适配器将执行结果ModelAndView返回给前端控制器。
  8. 调度器接收到ModelAndView后，请求对应的视图解析器ViewResolver。
  9. 视图解析器解析ModelAndView后返回对应视图View，将数据渲染到视图，并返回渲染后的视图给调度器
  10. 最后将渲染后的页面响应给用户或客户端

HelloController.java

public class HelloController implements Controller {
    @Override
    public ModelAndView handleRequest(HttpServletRequest httpServletRequest, HttpServletResponse httpServletResponse) throws Exception {
        ModelAndView modelAndView = new ModelAndView();
        // 调用业务层
        String result = "hellomvc";
        // 封装数据对象
        modelAndView.addObject("msg", result);
        // 封装视图名 会由视图解析器拼接成要跳转的视图 /WEB-INF/jsp/hello.jsp
        modelAndView.setViewName("hello");
        return modelAndView;
    }
}

请求映射

/**
 * @RequestMapping()
 *      作用：绑定 URI 和方法
 *      位置：
 *          1.类上              ip:port/ur
 *          2.方法上            ip:port/hello
 *          3.类 + 方法上       ip:port/ur/hello
 *      参数：
 *          value  URI请求路径
 *          method 请求类型
 *          params 请求路径参数  ip:port/ur?params
 */
@Controller
@RequestMapping(value = "/ur")
public class HelloController {
    // ur/hello?hh
    @RequestMapping(value = "/hello", method = {RequestMethod.GET}, params = "hh"))
    public ModelAndView hello() {
        ModelAndView mv = new ModelAndView();
        mv.addObject("hello", "Hello SpringMVC 绑定单个URI");
        mv.setViewName("hello");
        return mv;
    }
    // ur/a1?hh 或 ur/a2?hh
    @RequestMapping(value = {"/a1", "/a2"}, method = {RequestMethod.GET}, params = "hh")
    public ModelAndView helloArrayTest() {
        ModelAndView mv = new ModelAndView();
        mv.addObject("hello", "Hello SpringMVC 绑定多个URI");
        mv.setViewName("hello");
        return mv;
    }
}

传参

/**
 * 传参
 *      基本
 *          用包装类（避免500报错）
 *      引用
 *          String | Date | 数组 | JavaBean | 集合：List Set Map
 */
@Controller
public class ParamsController {
    /**
     * 基本
     *      用包装类（避免500报错）
     *      默认值null
     *      前端传参名与本方法参数名不一致时 使用 @RequestParam()
     *          参数
     *               name：别名
     *               defaultValue：默认值 
     */
    // p2?a=19&height=170.5
    @RequestMapping(value = "/p2")
    public void pt2(@RequestParam(name = "a", defaultValue = "18") Integer age, Double height) {
        System.out.println("age:" + age + ":height:" + height);
    }
    
    /**
    * 引用：数组
    */
    // p3?ids=1&ids=2&ids=3
    @RequestMapping("/p3")
    public void pt3(Integer[] ids) {
        for (Integer id : ids) {
            System.out.println(id);
        }
    }

    /**
     * 引用：String
     *      默认值为null
     */
    // p4?userName="zs"&userPwd="ss"
    @RequestMapping(value = "/p4")
    public void pt4(String userName, String userPwd) {
        System.out.println("userName:" + userName + ":userPwd:" + userPwd);
    }

    /**
     * JavaBean
     */
    // p5?id=1&userName="zs"&userPwd="ss"
    @RequestMapping(value = "/p5")
    public void pt5(User user){
        System.out.println(user);
    }

    /**
     * 引用：集合 List（用JavaBean进行包装）
     */
    // p6?userName=df&userPwd=lo&id=6&ids=1,2
    @RequestMapping(value = "/p6")
    public void pt6(User user){
        user.getIds().forEach(System.out::println);
    }

    /**
     * 引用：集合 Map（用JavaBean进行包装）
     */
    // p7?map['sh']=dfmz
    @RequestMapping(value = "/p7")
    public void pt7(User user){
        user.getMap().forEach((k,v)-> System.out.println("k:"+k+":v:"+v));
    }
}

    }

    @RequestMapping("m02")

        request.setAttribute("hello", "hello m02");
        return "hello";
    }

    // ModelMap（继承LinkedMap）
    @RequestMapping("m03")
    public String m03(ModelMap modelMap) {
        modelMap.addAttribute("hello", "hello m03");
        return "hello";
    }
    
    // Model  （ModelMap精简版）
	// 若返回值是String 且有对应视图 则直接传给视图解析器

        model.addAttribute("hello", "hello m04");
        // 返回视图名 同一个视图可复用（只写一个页面 传递不同的数据）
        return "hello";
    }

    @RequestMapping("m05")
    public String m05(Map map) {
        map.put("hello", "hello m05");
        return "hello";
    }
}

重定向 & 转发

/**
 * 重定向 转发
 * 		redirect:
 *		forward:
 */
@Controller
public class ViewController {

    // 重定向（不配置视图解析器）
    @RequestMapping(value = "/cdx")
    public String cdx(){
		return "redirect:/index.jsp";
    }
    
    // 重定向（视图解析器）
    @RequestMapping(value = "/cdx")
    public String cdx(){
		return "redirect:/index.jsp";
    }
    
    // 重定向 返回视图 带参
    @RequestMapping(value = "/r1")
    public String rt1(){
        return "redirect:/redirect?a=admin";
    }

    // 重定向 返回视图 中文
    @RequestMapping(value = "/r2")
    public String rt2(RedirectAttributes redirectAttributes){
        redirectAttributes.addAttribute("a","奥利给");
        return "redirect:/redirect";
    }

    // 重定向 返回模型和视图 中文
    @RequestMapping(value = "/r3")
    public ModelAndView rt3(){
        ModelAndView mv=new ModelAndView();
        mv.addObject("a","奥利给");
        mv.setViewName("redirect:/redirect");
        return mv;
    }

    // 转发（不配置视图解析器）
    @RequestMapping(value = "/zf")
    public String zf(){
		return "forward:/index.jsp";
    }
    
    // 转发（视图解析器）
    @RequestMapping(value = "/zf")
    public String zf(){
		return "index";
    }
    
    // 转发 到方法 返回模型和视图 带参
    @RequestMapping(value = "/z1")
    public ModelAndView zt1(){
        ModelAndView mv=new ModelAndView();
        mv.setViewName("forward:test?a=admin&b=1");
        return mv;
    }

    // 转发 到方法 返回模型和视图 带参 中文
    @RequestMapping(value = "/z2")
    public ModelAndView zt2(){
        ModelAndView mv=new ModelAndView();
        mv.addObject("a", "admin");
        mv.addObject("b", "奥利给");
        mv.setViewName("forward:test");
        return mv;
    }
}

RestFul 风格

🌠 概念：通过不同请求方式实现不同效果，实现URL的复用、访问安全。

⭐ 理解：请求地址一样，但功能不同。

效果 URL
增 POST http://127.0.0.1/item/
删 DELETE http://127.0.0.1/item/1
改 PUT http://127.0.0.1/item/
查 GET http://127.0.0.1/item/1 
@Controller
public class HelloController {
    // 非RestFul风格：http://localhost:8080/hello?a=1&b=haha
    // RestFul风格（约束请求类型为GET）：http://localhost:8080/hello/1/haha
    // 方法级别的注解变体：@PostMapping @DeleteMapping @PutMapping @GetMapping @PatchMapping
//    @RequestMapping(value = "/hello/{a}/{b}", method = RequestMethod.GET)
    @GetMapping(value = "/hello/{a}/{b}")
    public ModelAndView hello(@PathVariable int a, @PathVariable("b") String bb) {
        ModelAndView modelAndView = new ModelAndView();
        // 调用业务层
        String result = "hello MVC!" + a + bb;
        // 封装数据对象
        modelAndView.addObject("msg", result);
        // 封装视图名 会由视图解析器拼接成要跳转的视图 /WEB-INF/jsp/hello.jsp
        modelAndView.setViewName("hello");
        return modelAndView;
    }
}

拦截器

🌠 概念：Interceptor，类似于 Servlet 的过滤器 Filter 。
🎅 特点：
      1. AOP 思想的具体应用。
      2. 只拦截访问的控制器方法，自带静态资源过滤（不会拦截 .jsp .html .css .js image 等）。
📌 作用：拦截用户请求并进行相应的处理，如权限验证、判断用户是否登陆等。

实现

实现接口：org.springframework.web.servlet.HandlerInterceptor

实现接口

MyInterceptor1.java

public class MyInterceptor1 implements HandlerInterceptor {
    
    @Override
    public boolean preHandle(HttpServletRequest request, HttpServletResponse response, Object handler) throws Exception {
        System.out.println("目标Handler执行前 执行MyInterceptor1-->preHandle 方法...");
        // 允许执行目标方法  true：放行 false: 拦截
        return false;
    }
	// 可以写拦截日志
    @Override
    public void postHandle(HttpServletRequest request, HttpServletResponse response, Object handler, ModelAndView modelAndView) throws Exception {
        System.out.println("目标Handler执行后 视图生成前 执行MyInterceptor1-->postHandle 方法...");
    }
	// 可以写拦截日志
    @Override
    public void afterCompletion(HttpServletRequest request, HttpServletResponse response, Object handler, Exception ex) throws Exception {
        System.out.println("目标Handler执行后 视图生成后 执行MyInterceptor1-->afterCompletion 方法...");
    }
}

LoginInterceptor.java

public class LoginInterceptor extends HandlerInterceptorAdapter {
    /**
     *
     * 获取用户session 信息
     *      若用户session 信息存在  请求合法 放行
     *      若用户session 信息不存在 请求非法 阻止
     *
     * @param request
     * @param response
     * @param handler
     * @return
     * @throws Exception
     */
    @Override
    public boolean preHandle(HttpServletRequest request, HttpServletResponse response, Object handler) throws Exception {
        HttpSession session = request.getSession();
        // 用户未登录 或 session失效
        if (session.getAttribute("userInfo") == null) {
            response.sendRedirect(request.getContextPath() + "/login.jsp");
            return false;
        }
        return true;
    }
}

enctype
    application/x-www=form-urlencoded：将表单域中的value属性值处理成url编码（默认）。
    multipart/form-data：以二进制流处理表单数据，把文件域指定文件的内容封装到请求参数里，不会对字符编码。
    text/plain：除了把空格转换为+号外，其他字符不做处理（适合发送邮件）。

@Controller
public class FileController {
    // 单文件
    @RequestMapping("uploadFile")
    public ModelAndView uploadFile(HttpServletRequest request) {
        ModelAndView mv = new ModelAndView();
        mv.setViewName("result");
        MultipartHttpServletRequest multipartHttpServletRequest = (MultipartHttpServletRequest) request;
        // MultipartFile 类用来接收前台传来的文件
        MultipartFile multipartFile = multipartHttpServletRequest.getFile("file");
        try {
            // 判断上传的文件、文件内容是否存在
            if (null != multipartFile && multipartFile.getSize() > 0) {
                // 获取项目绝对路径  D:\Java\JavaProject\SpringWebTest\src\main\webapp\
//                String basePath1 = request.getSession().getServletContext().getRealPath("/");
                String basePath = request.getServletContext().getRealPath("/");
                // 获取文件夹对象
                File file = new File(basePath + "/upload");
                if (!(file.exists())) {
                    // 文件夹不存在 创建文件夹
                    boolean mkdirResult = file.mkdir();
                    if (!mkdirResult) {
                        mv.addObject("msg", "文件夹创建失败，文件上传失败");
                    }
                }
                // 设置文件名  系统当前时间毫秒数+命名上传文件  getOriginalFilename() 原始文件名
                String fileName = System.currentTimeMillis() + multipartFile.getOriginalFilename().substring(multipartFile.getOriginalFilename().lastIndexOf("."));
                // 保存文件 文件copy
                multipartFile.transferTo(new File(file, fileName));
                mv.addObject("msg", "文件上传成功");
            } else {
                mv.addObject("msg", "文件上传失败");
            }
        } catch (Exception e) {
            e.printStackTrace();
            mv.addObject("msg", "文件上传失败");
        }
        return mv;
    }

    // 多文件
    @RequestMapping("uploadFiles")
    public ModelAndView uploadFiles(HttpServletRequest request) {
        ModelAndView mv = new ModelAndView();
        mv.setViewName("result");
        MultipartHttpServletRequest multipartHttpServletRequest = (MultipartHttpServletRequest) request;
        List<MultipartFile> multipartFiles = multipartHttpServletRequest.getFiles("file");
        StringBuilder stringBuilder = new StringBuilder();
        for (MultipartFile multipartFile : multipartFiles) {
            try {
                if (multipartFile.getSize() > 0) {
                    String basePath = request.getServletContext().getRealPath("/");
                    File file = new File(basePath + "/upload");
                    if (!(file.exists())) {
                        if (!(file.mkdir())) {
                            mv.addObject("msg", "文件上传失败：文件夹创建失败");
                        }
                    }
                    String originalFileName = "文件：" + multipartFile.getOriginalFilename() + "&nbsp;&nbsp;上传成功<br/>";
                    stringBuilder.append(originalFileName);
                    String fileName = System.currentTimeMillis() + multipartFile.getOriginalFilename().substring(multipartFile.getOriginalFilename().lastIndexOf("."));
                    multipartFile.transferTo(new File(file, fileName));
                } else {
                    String originalFileName = "文件上传失败：请选择至少一个文件<br/>";
                    stringBuilder.append(originalFileName);
                }
            } catch (Exception e) {
                e.printStackTrace();
                mv.addObject("msg", "文件上传失败：异常");
            
        }
        mv.addObject("msg", stringBuilder);
        return mv;
    }
        
    // 单文件2
    // @RequestParam("file") 将name=file控件得到的文件封装为commonsMultipartFile对象
    // 批量上传commonsMultipartFile则为数组即可
    @RequestMapping("/uploadFile2")
    public String uploadFile2(@RequestParam("file") CommonsMultipartFile commonsMultipartFile, HttpServletRequest request) throws IOException {
        // 上传路径
        String basePath = request.getServletContext().getRealPath("/");
        File file = new File(basePath + "/upload");
        if (!(file.exists())) {
            // 文件夹不存在 创建文件夹
            boolean mkdirResult = file.mkdir();
            if (!mkdirResult) {
                mv.addObject("msg", "文件夹创建失败，文件上传失败");
            }
        }
        commonsMultipartFile.transferTo(new File(file + "/" + commonsMultipartFile.getOriginalFilename()));
        return "redirect:/index.jsp";
    }
}

/**
 * 文件下载
 *
 *  1.设置响应类型 （浏览器无法使用某种方式或激活某个程序来处理的 MIME 类型）
 *    	   response.setContentType("application/x-msdownload");
 *    	或 response.setContentType("application/octet-stream");
 *    设置响应头
 *   	 response.setHeader("Content-Disposition", "attachment;filename=" + fileName);
 *  2.读取所需文件（文件输入流 new FileInputStream(file)）
 *  3.响应（字节输出流 response.getOutputStream().write() ）
 *  4.关闭 文件输入流、字节输出流
 */
	@RequestMapping("/downloadFile")
	public String downloadFile (HttpServletRequest request, HttpServletResponse response) throws Exception{
        // 获取参数 （获取要下载的文件名）
        String fileName = request.getParameter("fileName");
        // 非空判断
        if (fileName == null || "".equals(fileName.trim())) {
            response.getWriter().write("请输入要下载的文件名！");
            response.getWriter().close();
            return;
        }
        // 得到资源的绝对路径
        String basePath = request.getServletContext().getRealPath("/download/");
        // 获取资源file对象
        File file = new File(basePath + fileName);
        // 判断文件对象是否存在并且是一个标准文件
        if (file.exists() && file.isFile()) {
            // 响应类型
            response.reset();  // 页面不缓存 清空buffer
            response.setCharacterEncoding("UTF-8");  // 字符编码
            response.setContentType("multipart/form-data");  // 二进制传输数据
            //response.setContentType("application/x-msdownload");
            //response.setContentType("application/octet-stream");
            // 响应头
            response.setHeader("Content-Disposition", "attachment;filename=" + fileName);
            // 获取文件输入流
            InputStream in = new FileInputStream(file);
            // 获取字节输出流
            ServletOutputStream out = response.getOutputStream();
            int len;
            byte[] bytes = new byte[1024];
            // 循环输出
            while ((len = in.read(bytes)) != -1){
                out.write(bytes,0,len);
                out.flush();  // 新增
            }
            // 关闭输入输出流
            out.close();
            in.close();
        } else {
            response.getWriter().write("文件不存在，请重试！");
            response.getWriter().close();
        }
        return null;
    }
}
