Swagger 快速入门

Java SpringBoot Swagger
Java
发布日期:   2020-05-02
文章字数:   851
阅读时长:   4 分

原生 UI

我们将使用 SpringBoot 集成 Swagger 2 ，快速体验。

环境

导入 Swager 2 依赖：

pom.xml

<dependency>
    <groupId>io.springfox</groupId>
    <artifactId>springfox-swagger2</artifactId>
    <version>2.9.2</version>
</dependency>
<dependency>
    <groupId>io.springfox</groupId>
    <artifactId>springfox-swagger-ui</artifactId>
    <version>2.9.2</version>
</dependency>

测试

1⃣ 创建 Swagger 2 配置文件

开启 Swagger 2 注解：@EnableSwagger2 ，因为设置了分环境，所有 Swagger 2 只在项目的测试环境中生效。

application.yaml

server:
  port: 8080
# 多环境 自定义激活指定配置文件
spring:
  profiles:
    active: dev

Swagger2Config.java

import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.core.env.Environment;
import org.springframework.core.env.Profiles;
import springfox.documentation.builders.RequestHandlerSelectors;
import springfox.documentation.service.ApiInfo;
import springfox.documentation.service.Contact;
import springfox.documentation.spi.DocumentationType;
import springfox.documentation.spring.web.plugins.Docket;
import springfox.documentation.swagger2.annotations.EnableSwagger2;

import java.util.ArrayList;

@Configuration
@EnableSwagger2
public class SwaggerConfig {
    @Bean
    public Docket docket(Environment environment) {
        // 分环境启用Swagger2
        Profiles profiles = Profiles.of("dev","test");
        boolean flag = environment.acceptsProfiles(profiles);
        return new Docket(DocumentationType.SWAGGER_2)
                // 是否启用Swagger2
                .enable(flag)
                .apiInfo(apiInfo())  // API信息
                // 接口扫描路径
                //  basePackage() 扫描指定包
                //  any()  扫描所有
                //  none()  都不扫描
                //  withClassAnnotation()  扫描类上的注解 Controller.class
                //  withMethodAnnotation()  扫描方法上的注解 GetMapping.class
                .select()
                .apis(RequestHandlerSelectors.basePackage("com.revelvy.controller"))
                // 接口过滤路径（排除）
//                .paths(PathSelectors.ant("/revelvy/**"))
                .build();
    }

    private ApiInfo apiInfo() {
        return new ApiInfo(
                "我的API文档",
                "我是API文档的描述",
                "v1.0",
                "https://www.revelvy.cn",
                new Contact(
                        "Revelvy",
                        "https://www.blog.revelvy.cn",
                        "xxx@qq.com"),
                "Apache 2.0",
                "http://www.apache.org/licenses/LICENSE-2.0",
                new ArrayList<>());
    }
}

2⃣ ​​自定义 API 文档注释（可选）

API 文档注释 作用
@ApiModel() 实体类模型
@ApiModelProperty() 实体类模型字段
@ApiOperation() 接口方法
@ApiParam() 接口方法形参

User.java

@ApiModel("用户实体类")
public class User {
    @ApiModelProperty("用户ID")
    private int id;
}

3⃣ 启动

此时，启动 Spring Boot 工程，在浏览器访问：http://localhost:8080/swagger-ui.html。

第三方 UI

Knife4j 是为 Java MVC 框架集成 Swagger 生成Api文档的增强解决方案，前身是swagger-bootstrap-ui，功能更加强悍！

Github 地址：https://github.com/xiaoymin/swagger-bootstrap-ui

官方文档：https://doc.xiaominfo.com/knife4j/documentation/

我这里带大家快速上手体验一番：

环境

导入 knife4j-spring-boot-starter 依赖，已自带 Swagger 2：

pom.xml

<dependency>
    <groupId>com.github.xiaoymin</groupId>
    <artifactId>knife4j-spring-boot-starter</artifactId>
    <version>3.0.3</version>
</dependency>

可根据需要开启增强配置：

application.yml

server:
  # 端口
  port: 8080
knife4j:
  # 开启增强配置
  enable: true
  # 开启生产环境屏蔽
  production: true
  # 开启Swagger的Basic认证功能,默认是 false
  basic:
    enable: true
    # Basic认证用户名
    username: test
    # Basic认证密码
    password: 123

测试

创建 Swagger 2 配置文件：

Swagger2Config

import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import springfox.documentation.builders.ApiInfoBuilder;
import springfox.documentation.builders.PathSelectors;
import springfox.documentation.builders.RequestHandlerSelectors;
import springfox.documentation.service.ApiInfo;
import springfox.documentation.service.ApiKey;
import springfox.documentation.service.AuthorizationScope;
import springfox.documentation.service.Contact;
import springfox.documentation.service.SecurityReference;
import springfox.documentation.spi.DocumentationType;
import springfox.documentation.spi.service.contexts.SecurityContext;
import springfox.documentation.spring.web.plugins.Docket;
import springfox.documentation.swagger2.annotations.EnableSwagger2;

import java.util.ArrayList;
import java.util.List;

/**
 * Swagger2 配置类
 *
 * @author revelvy
 * @since 1.0.0
 */
@Configuration
@EnableSwagger2
public class Swagger2Config {

    @Bean
    public Docket createRestApi() {
        return new Docket(DocumentationType.SWAGGER_2)
                .apiInfo(apiInfo())
                .select()
                // 为所有接口生成API文档
//                .apis(RequestHandlerSelectors.any())
                .apis(RequestHandlerSelectors.basePackage("com.xxxx.server.controller"))
                .paths(PathSelectors.any())
                .build();
    }

    private ApiInfo apiInfo() {
        return new ApiInfoBuilder()
                .title("xxx接口文档")
                .description("xxx接口文档")
                .termsOfServiceUrl("http://www.xx.com/")
                .contact(new Contact("Revelvy", "", "xxx@qq.com"))
                .version("v1.0")
                .build();
    }
}

此时，启动 Spring Boot 工程，在浏览器访问：http://localhost:8080/doc.html
